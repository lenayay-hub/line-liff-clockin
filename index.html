  <script>

    window.vConsole = new window.VConsole();

    console.log("📸 vConsole 啟動！");


    // ==========================================

    // ⚠️ 關鍵填入：

    // 1. 你的 LINE LIFF ID

    // 2. 剛剛第一步部署完獲得的 GAS 網頁應用程式網址 (結尾必須是 /exec)

    // ==========================================

    const MY_LIFF_ID = '2010271306-bjI5SC09'; 

    const GAS_API_URL = 'https://script.google.com/macros/s/AKfycbwsfDTBlKMjF1FgX_PRLKXnNDgZQtjcjKqY25aDLkpUtp6BuZwg3qWsmzD9vImOFHyq/exec'; 

    // ==========================================


    var globalLineUserId = ""; 

    var cachedUnboundList = []; 


    // 通用網路要求傳輸大腦：避開 CORS 預檢限制

    function callGasApi(action, payload) {

      console.log(`📡 發送 API 請求 -> 指令: ${action}`, payload);

      return fetch(GAS_API_URL, {

        method: "POST",

        body: JSON.stringify({ action: action, ...payload }),

        headers: { "Content-Type": "text/plain" } 

      })

      .then(res => {

        if (!res.ok) throw new Error("網路連線異常，伺服器無回應");

        return res.json();

      });

    }


    document.addEventListener("DOMContentLoaded", function() {

      showStatus("正在初始化 LINE 安全通道...", "loading");

      

      liff.init({ liffId: MY_LIFF_ID })

        .then(() => {

          console.log("✅ LIFF SDK 在獨立網頁環境下順利啟動！");

          if (!liff.isLoggedIn()) {

            liff.login(); 

          } else {

            return liff.getProfile(); 

          }

        })

        .then((profile) => {

          if (!profile) return;

          globalLineUserId = profile.userId;

          console.log("👤 成功取得 LINE UID:", globalLineUserId);

          

          document.getElementById("liffStatus").innerText = "LINE 驗證成功，正在檢查綁定狀態...";

          

          // 改用 fetch 呼叫後端 API

          return callGasApi("checkLineBinding", { lineUserId: globalLineUserId });

        })

        .then(handleBindingResult)

        .catch((err) => {

          console.error("💥 核心管道錯誤:", err);

          showStatus("核心連線失敗，請檢查網路或系統設定。錯誤訊號: " + err.message, "error");

          document.getElementById("liffStatus").innerText = "連線建立失敗。";

        });

    });


    function handleBindingResult(result) {

      console.log("📥 成功接收後端回傳封包:", result);

      if (!result.success) {

        showStatus("資料庫通道故障: " + result.message, "error");

        return;

      }


      hideStatus();

      const nameSelect = document.getElementById("nameSelect");

      nameSelect.innerHTML = ""; 


      if (result.isBound) {

        document.getElementById("liffStatus").innerText = `員工連線成功！歡迎回來，${result.name} 👋`;

        

        let option = document.createElement("option");

        option.value = result.name;

        option.text = result.name;

        nameSelect.appendChild(option);

        nameSelect.disabled = true;


        document.getElementById("deptInput").value = result.department;

        document.getElementById("salaryTypeInput").value = result.salaryType;


        document.getElementById("clockSection").style.display = "block";

        document.getElementById("bindSection").style.display = "none";

      } else {

        document.getElementById("liffStatus").innerText = "偵測到新裝置！請選擇您的姓名進行帳號綁定。";

        cachedUnboundList = result.unboundList;


        if (!cachedUnboundList || cachedUnboundList.length === 0) {

          let option = document.createElement("option");

          option.value = "";

          option.text = "目前沒有未綁定的員工名單";

          nameSelect.appendChild(option);

          showStatus("後端無未綁定同仁名單，請管理員確認試算表資訊。", "error");

          return;

        }


        let defaultOption = document.createElement("option");

        defaultOption.value = "";

        defaultOption.text = "-- 請選擇您的姓名 --";

        nameSelect.appendChild(defaultOption);


        cachedUnboundList.forEach(emp => {

          let option = document.createElement("option");

          option.value = emp.name;

          option.text = emp.name;

          nameSelect.appendChild(option);

        });


        document.getElementById("bindSection").style.display = "block";

        document.getElementById("clockSection").style.display = "none";

      }

    }


    function autoFillEmployeeInfo() {

      const selectedName = document.getElementById("nameSelect").value;

      const deptInput = document.getElementById("deptInput");

      const salaryTypeInput = document.getElementById("salaryTypeInput");


      if (!selectedName) {

        deptInput.value = "";

        salaryTypeInput.value = "";

        return;

      }


      const emp = cachedUnboundList.find(e => e.name === selectedName);

      if (emp) {

        deptInput.value = emp.department;

        salaryTypeInput.value = emp.salaryType;

      }

    }


    function submitBinding() {

      const selectedName = document.getElementById("nameSelect").value;

      if (!selectedName) {

        alert("請先選取您的姓名！");

        return;

      }


      showStatus("正在寫入試算表並綁定 LINE 帳號...", "loading");

      document.getElementById("bindBtn").disabled = true;


      callGasApi("linkEmployeeLineId", { name: selectedName, lineUserId: globalLineUserId })

        .then(result => {

          if (result.success) {

            showStatus(result.message, "success");

            setTimeout(() => { location.reload(); }, 2000);

          } else {

            showStatus(result.message, "error");

            document.getElementById("bindBtn").disabled = false;

          }

        })

        .catch(handleGasError);

    }


    function startClockProcess(clockType) {

      showStatus("🔍 正在透過手機 GPS 獲取精確定位...", "loading");

      setClockButtonsDisabled(true);


      if (!navigator.geolocation) {

        showStatus("您的手機裝置不支援 GPS 定位功能。", "error");

        setClockButtonsDisabled(false);

        return;

      }


      navigator.geolocation.getCurrentPosition(

        function(position) {

          const lat = position.coords.latitude;

          const lng = position.coords.longitude;

          console.log(`📍 定位成功: (${lat}, ${lng})`);

          

          showStatus(`經緯度讀取成功，正在送出打卡紀錄...`, "loading");


          const clockPackage = {

            name: document.getElementById("nameSelect").value,

            department: document.getElementById("deptInput").value,

            type2: document.getElementById("salaryTypeInput").value,

            clockType: clockType,

            lat: lat,

            lng: lng,

            note: document.getElementById("noteInput").value

          };


          callGasApi("saveClock", { data: clockPackage })

            .then(result => {

              if (result.success) {

                showStatus(result.message, "success");

                document.getElementById("noteInput").value = ""; 

              } else {

                showStatus(result.message, "error");

              }

              setClockButtonsDisabled(false);

            })

            .catch(handleGasError);

        },

        function(error) {

          let errorMsg = "定位失敗：";

          switch(error.code) {

            case error.PERMISSION_DENIED:

              errorMsg += "請允許 LINE 應用程式存取您的「位置權限」。";

              break;

            case error.POSITION_UNAVAILABLE:

              errorMsg += "無法取得位置訊號，請至通訊良好的地方重試。";

              break;

            case error.TIMEOUT:

              errorMsg += "搜尋 GPS 逾時，請再試一次。";

              break;

            default:

              errorMsg += error.message;

          }

          showStatus(errorMsg, "error");

          setClockButtonsDisabled(false);

        },

        { enableHighAccuracy: true, timeout: 15000, maximumAge: 0 }

      );

    }


    function setClockButtonsDisabled(disabled) {

      document.getElementById("btnIn").disabled = disabled;

      document.getElementById("btnOut").disabled = disabled;

    }


    function handleGasError(err) {

      showStatus("資料傳輸故障：" + err.toString(), "error");

      document.getElementById("bindBtn").disabled = false;

      setClockButtonsDisabled(false);

    }


    function showStatus(msg, type) {

      const box = document.getElementById("statusBox");

      box.innerText = msg;

      box.className = "status-box status-" + type;

      box.style.display = "block";

    }


    function hideStatus() {

      document.getElementById("statusBox").style.display = "none";

    }
